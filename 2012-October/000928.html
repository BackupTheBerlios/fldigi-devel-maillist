A small bulb ... I build on Ubuntu 11.10 ... no change in mingw32 
environment since change to fltk-1.3.0

dave at linux-dev:~/dev.git/all$ gcc --version
gcc (Ubuntu/Linaro 4.6.1-9ubuntu3) 4.6.1

dave at linux-dev:~/dev.git/all$ echo $CROSSCFG
--build=i686-pc-linux-gnu --host=i586-mingw32msvc

dave at linux-dev:~/dev.git/all$ echo $PKGCFG
PKG_CONFIG=/usr/local/win32/bin/i586-mingw32msvc-pkg-config

dave at linux-dev:~/dev.git/all$ echo $PREFIX
/usr/local/win32

dave at linux-dev:~/dev.git/all$ /usr/local/win32/bin/fltk-config --version
1.3.0

dave at linux-dev:~/dev.git/all$ ./configure $CROSSCFG $PKGCFG 
FLTK_CONFIG=$PREFIX/bin/fltk-config --with-ptw32=$PREFIX/ptw32 
XMLRPC_C_CONFIG=$PREFIX/bin/xmlrpc-c-config

dave at linux-dev:~/dev.git/all$ make

dave at linux-dev:~/dev.git/all$ i586-mingw32msvc-strip src/flmsg.exe

dave at linux-dev:~/dev.git/all$ make nsisinst

dave at linux-dev:~/dev.git/all$ mv src/*setup*exe .
